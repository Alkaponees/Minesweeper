#include<SFML/Graphics.hpp>
#include<iostream>
#include<time.h>
#include<stdlib.h>
using namespace std;
using namespace sf;
int main()
{//1.Create a menu
	RenderWindow app(VideoMode(400,400),"Minesweeper!");//розмір окна і назва, яка буде відображатися.
	while (app.isOpen())
	{
		Event e;
		while (app.pollEvent(e))
		{
			if (e.type == Event::Closed)
			{
				app.close();
			}
		}

	}

	return 0;
}
